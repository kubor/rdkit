244 //! functor to compare two doubles with a tolerance
245 struct ltDouble {
246  public:
247  ltDouble() : _tol(1.0e-8){};
248  bool operator()(double d1, double d2) const {
249  if (fabs(d1 - d2) < _tol) {
250  return false;
251  } else {
252  return (d1 < d2);
253  }
254  }
255 
256  private:
257  double _tol;
258 };
