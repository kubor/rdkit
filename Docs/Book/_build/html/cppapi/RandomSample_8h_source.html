43 //! This is a class for fully randomly sampling reagents.
44 // Note that this enumerator never halts.
45 /*!
46  basic usage:
47 
48  \verbatim
49  std::vector<MOL_SPTR_VECT> bbs;
50  bbs.push_back( bbs_for_reactants_1 );
51  bbs.push_back( bbs_for_reactants_2 );
52 
53  RandomSampleStrategy rgroups;
54  rgroups.initialize(rxn, bbs);
55  for(size_t i=0; i<num_samples && rgroups; ++i) {
56  MOL_SPTR_VECT rvect = getReactantsFromRGroups(bbs, rgroups.next());
57  std::vector<MOL_SPTR_VECT> lprops = rxn.RunReactants(rvect);
58  ...
59  }
60  \endverbatim
61 
62  See EnumerationStrategyBase for more details and usage.
63 */
64 class RandomSampleStrategy : public EnumerationStrategyBase {
65  boost::uint64_t m_numPermutationsProcessed;
66  boost::minstd_rand m_rng;
67  std::vector<boost::random::uniform_int_distribution<> > m_distributions;

69  public:
70  RandomSampleStrategy()

75  for (size_t i = 0; i < m_permutation.size(); ++i) {

77  boost::random::uniform_int_distribution<>(0, m_permutation[i] - 1));
