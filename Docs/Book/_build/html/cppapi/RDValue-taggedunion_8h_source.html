54 // RDValue does not dynamically create POD types (kind of like
55 // cdiggins::any) However, it doesn't use RTTI type info
56 // directly, it uses a companion short valued type
57 // to determine what to do.
58 // For unregistered types, it falls back to boost::any.
59 // The Size of an RDAny is (sizeof(double) + sizeof(short) == 10 bytes
60 // (aligned to actually 16 so hard to pass as value type)
61 //
62 // For the sake of compatibility, errors throw boost::bad_any_cast
63 //
64 // Examples:
65 //
66 // RDAny v(2.);
67 // v = 1;
68 // std::vector<double> d;
69 // v == d;
70 // v.asDoubleVect().push_back(4.)
71 // rdany_cast<std::vector<double>(v).push_back(4.)
72 //
73 // Falls back to boost::any for non registered types
74 // v = boost::shared_ptr<ROMol>(new ROMol(m));
75 //
76 
77 // RDValue does not manange memory of non-pod data
78 // this must be done externally (string, Any, vector...)
79 // Tagged union

260  // RDValue doesn't have an explicit destructor, it must
261  // be wrapped in a container.
262  // The idea is that POD types don't need to be destroyed
263  // and this allows the container optimization possibilities.
264  void destroy() {
265  switch (type) {
266  case RDTypeTag::StringTag:
267  delete value.s;
268  break;
269  case RDTypeTag::AnyTag:
270  delete value.a;
271  break;
272  case RDTypeTag::VecDoubleTag:
273  delete value.vd;
274  break;
275  case RDTypeTag::VecFloatTag:
276  delete value.vf;
277  break;
278  case RDTypeTag::VecIntTag:
279  delete value.vi;
280  break;
281  case RDTypeTag::VecUnsignedIntTag:
282  delete value.vu;
283  break;
284  case RDTypeTag::VecStringTag:
285  delete value.vs;
286  break;
287  default:
288  break;
289  }
290  type = RDTypeTag::EmptyTag;
291  }

301 // Given two RDValue::Values - copy the appropriate structure
302 // RDValue doesn't have a copy constructor, the default
303 // copy act's like a move for better value semantics.
304 // Containers may need to copy though.
305 inline void copy_rdvalue(RDValue &dest, const RDValue &src) {
306  if (&dest == &src) // don't copy over yourself
307  return;
308  dest.destroy();
309  dest.type = src.type;
310  switch (src.type) {
311  case RDTypeTag::StringTag:
312  dest.value.s = new std::string(*src.value.s);
313  break;
314  case RDTypeTag::AnyTag:
315  dest.value.a = new boost::any(*src.value.a);
316  break;
317  case RDTypeTag::VecDoubleTag:
318  dest.value.vd = new std::vector<double>(*src.value.vd);
319  break;
320  case RDTypeTag::VecFloatTag:
321  dest.value.vf = new std::vector<float>(*src.value.vf);
322  break;
323  case RDTypeTag::VecIntTag:
324  dest.value.vi = new std::vector<int>(*src.value.vi);
325  break;
326  case RDTypeTag::VecUnsignedIntTag:
327  dest.value.vu = new std::vector<unsigned int>(*src.value.vu);
328  break;
329  case RDTypeTag::VecStringTag:
330  dest.value.vs = new std::vector<std::string>(*src.value.vs);
331  break;
332  default:
333  dest = src;
334  }
335 }

360 // Get stuff stored in boost any
361 template <class T>
362 inline T rdvalue_cast(RDValue_cast_t v) {
363  // Disable reference and pointer casts to POD data.
364  BOOST_STATIC_ASSERT(!(
365  (boost::is_pointer<T>::value &&
366  (boost::is_integral<typename boost::remove_pointer<T>::type>::value ||
367  boost::is_floating_point<
368  typename boost::remove_pointer<T>::type>::value)) ||
369  (boost::is_reference<T>::value &&
370  (boost::is_integral<typename boost::remove_reference<T>::type>::value ||
371  boost::is_floating_point<
372  typename boost::remove_reference<T>::type>::value))));
373 
374  if (rdvalue_is<boost::any>(v)) {
375  return boost::any_cast<T>(*v.ptrCast<boost::any>());
376  }
377  throw boost::bad_any_cast();
378 }
