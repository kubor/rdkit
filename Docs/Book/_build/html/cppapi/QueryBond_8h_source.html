95 namespace detail {
96 inline std::string qhelper(Bond::QUERYBOND_QUERY *q, unsigned int depth) {
97  std::string res;
98  if (q) {
99  for (unsigned int i = 0; i < depth; ++i) res += " ";
100  res += q->getFullDescription() + "\n";
101  for (Bond::QUERYBOND_QUERY::CHILD_VECT_CI ci = q->beginChildren();
102  ci != q->endChildren(); ++ci) {
103  res += qhelper((*ci).get(), depth + 1);
104  }
105  }
106  return res;
107 }
108 } // end of detail namespace
109 inline std::string describeQuery(const Bond *bond) {
